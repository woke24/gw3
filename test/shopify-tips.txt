45. gsap
<script>
  document.addEventListener("DOMContentLoaded", function () {
    gsap.registerPlugin(ScrollTrigger);
    gsap.from(".ic-item", {
      scrollTrigger: {
        trigger: ".ic-item", // 目标 Section
        start: "top 80%",  // 当元素到达视口顶部80% 时触发， center center为在视图中心位置，
        toggleActions: "play none none reset", // 每次进入视口时重新执行

        {% comment %}
         toggleActions：
         onEnter - 当滚动进入触发点时
         onLeave - 当滚动离开触发点时
         onEnterBack - 当从下方反向滚动回到触发点时
         onLeaveBack - 当从上方反向滚动离开触发点时

          "play" - 播放动画
          "pause" - 暂停动画
          "resume" - 继续动画
          "reset" - 重置动画到初始状态
          "restart" - 重新开始动画
          "complete" - 直接完成动画
          "reverse" - 反向播放动画
          "none" - 不执行任何操作
         {% endcomment %}
      },
      opacity: 0,// 初始透明度为0（从完全透明开始）
      y: 50,// 初始位置在下方50像素(x为水平)
      duration: 0.5,// 动画持续1秒
      {% comment %} stagger: 0.2 // 依次触发  {% endcomment %}
    });
  });
</script>

start: "bottom bottom-=100", 动画区间距上下各100， 不设置默认底部
 end: "top top+=100",不设置默认顶部
toggleActions: "play none none reverse",进入正放，离开倒放
